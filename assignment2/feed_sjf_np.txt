void feed_sjf_np(CLL *incoming_job_queue)
{
    /* Initiate a job list
    * Loop1: Start a loop which runs until dispatcher has no more jobs to dispatch
        * Is a job finishing this instant?
            * Yes, then remove it from pending_jobs_queue
                * update this time as ending of the job to help with the Gantt chart creation 
                * Since this is non pre-emptive we don't need to re-sort the pending_jobs_queue
                * Is a job arriving this instant?
                    * Yes, 
                        * then put the job at its rightful place in the sorted pending_jobs_queue
                    * No, 
                        * continue from Loop1
            * No, 
                * Is a job arriving this instant?
                    * Yes,
                        * then put the job at its rightful place in the sorted pending_jobs_queue
                    * Is a job running already?
                        * Yes, 
                            * increment its run_time, clock
                            * continue from Loop1
                        * No, 
                            * pick the job at the beginning of the pending_jobs_queue as the current job
                            * remove the job from pending_jobs_queue
                            * start the job and update its start_time with scheduler clock ticks at the instant
                            * increment its run_time, clock
                            * continue from Loop1
    */
    return;   
}
